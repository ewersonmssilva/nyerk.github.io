function OnRun()
	local CurAdmaProc = FrameworkExp.procedure[FrameworkExp:ExternalCurrentProcedureInQueue()];
	
	if CurAdmaProc[CurAdmaProc.idx] == "ExternalProcedureKey" then
		Write("ExternalProcedureKey");
		local procToPush = "";
		
		if InventoryList:GetInventory("Observation Post Passage Key") ~= nil then
		--	procToPush = baseProcName.. "-ObservationPostPassageKey";
		elseif InventoryList:GetInventory("Great Dining Hall Key") ~= nil then
			procToPush = baseProcName.. "-GreatDiningHallKey";
		elseif InventoryList:GetInventory("Treasury Key") ~= nil then
			procToPush = baseProcName.. "-TreasuryKey";
		elseif InventoryList:GetInventory("Servants Quarters Key") ~= nil then
			procToPush = baseProcName.. "-ServantsQuartersKey";
		end
		
		CurAdmaProc.idx = CurAdmaProc.idx + 1;
		CurAdmaProc.tempIdx = CurAdmaProc.idx;
		
		if procToPush ~= "" then
			FrameworkExp:ExternalPushProcedure(procToPush);
		end
	end
	
	
	
	if not Player:IsDead() and not FrameworkExp:AdmaStrongholdSpecialBehaviour() then
		return false;
	end
	
	FrameworkExp:OnRun();
end

function OnFrame()
	FrameworkExp:OnFrame();
