Code: Select all

	while DialogList:GetDialog( "support_shortcut_dialog/skill_cond_1" ):IsVisible() do
		if PetBuff ~= 1 then
		DialogList:GetDialog( "support_shortcut_dialog/skill_cond_1" ):Click();
		PetBuff = 1;
		
		elseif PetBuff == 1 then
		PetBuff = 0;
		
		end
		end
